Phase 10 — Kubernetes on Docker Desktop

Prerequisite: Enable Kubernetes in Docker Desktop → Settings → Kubernetes → "Enable Kubernetes" → Apply & Restart (do this before recording; it takes 2-3 minutes)


# 1. Verify kubectl is pointing at Docker Desktop's cluster
kubectl config use-context docker-desktop
kubectl cluster-info

# 2. Confirm the image exists locally (built in Phase 5/6)
docker images heart-disease-api:latest

# 3. Deploy to Kubernetes
kubectl apply -f k8s/deployment.yaml
kubectl apply -f k8s/service.yaml

# 4. Watch pods come up
kubectl get pods -w
# Press Ctrl+C once both pods are Running

# 5. Check the service (LoadBalancer gets a localhost IP on Docker Desktop)
kubectl get svc heart-api
# Docker Desktop assigns localhost:80 for LoadBalancer services

# 6. Smoke test via Kubernetes
curl -s http://localhost/health | jq
curl -s -X POST http://localhost/predict \
  -H "Content-Type: application/json" \
  -d '{"age":63,"sex":1,"cp":4,"trestbps":145,"chol":233,"fbs":1,"restecg":2,"thalach":150,"exang":0,"oldpeak":2.3,"slope":3,"ca":0,"thal":6}' | jq

# 7. Check logs
kubectl logs -l app=heart-api --tail=50

# 8. (Optional) Deploy via Helm instead of raw kubectl
# helm install heart-api ./k8s/helm/heart-api

# 9. Clean up when done
kubectl delete -f k8s/deployment.yaml
kubectl delete -f k8s/service.yaml
